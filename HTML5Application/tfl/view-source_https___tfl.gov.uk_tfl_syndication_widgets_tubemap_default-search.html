function autoSuggest(event,id){
// catch backspaces
if(event.keyCode == '8'){
// delete everything if you clear the field, just to be safe
if(currentPoint==0){
currentPoint=0;
id.value='';
} else {
currentPoint=id.value.length;
}
// catch enter
} else if(event.keyCode == '13'){
if(document.selection){
// IE
var range = id.createTextRange();
range.collapse(true);
range.moveEnd('character', id.value.length);
range.moveStart('character', 0);
range.select();
// everyone else
} else {
id.setSelectionRange(0,id.value.length);
}
// catch end key
} else if(event.keyCode == '35'){
if(document.selection){
// IE
var range = id.createTextRange();
range.collapse(true);
range.moveEnd('character', id.value.length);
range.moveStart('character', id.value.length);
range.select();
} else {
id.setSelectionRange(id.value.length,id.value.length);
}
// check for 'st' entries without full stop
} else if (id.value.toLowerCase() == 'st. ' || id.value.toLowerCase() == 'saint '){
id.value = 'st ';
if(document.selection){
// IE
var range = id.createTextRange();
range.collapse(true);
range.moveEnd('character', id.value.length);
range.moveStart('character', id.value.length);
range.select();
} else {
id.setSelectionRange(id.value.length,id.value.length);
}
currentPoint = 4;
// else real characters
} else {
currentPoint++;
//check to see whether we're starting autocomplete
if(id.value.length > letterCount){
// loop through array and search
for(i=0;i<stationNames.length;i++){
// convert to lowercase for matches
stationName=stationNames[i][0].toLowerCase();
if(id.value.toLowerCase() == stationName.substring(0,id.value.length)){
id.value=stationName;
if(document.selection){
// IE
var range = id.createTextRange();
range.collapse(true);
range.moveEnd('character', id.value.length);
range.moveStart('character', currentPoint);
range.select();
} else {
id.setSelectionRange(currentPoint,id.value.length);
}
targetCoordX=stationNames[i][1];
targetCoordY=stationNames[i][2];
targetBoxLeft=stationNames[i][1];
targetBoxTop=stationNames[i][2];
//#####
targetBoxWidth = stationNames[i][3]; //-stationNames[i][1];

allowSearch=true;

break;
}
}
} else {
currentPoint=id.value.length;
}

function moveTo(){
// check for allowSearch and also ensure the station name hasn't changed
if(allowSearch==true && stationNames[foundArrayId][0].toLowerCase()==document.getElementById('searchBox').value.toLowerCase()){
coordX=targetCoordX;
coordY=targetCoordY;
mapWidth=parseInt(document.getElementById('map-medium').width);
mapHeight=parseInt(document.getElementById('map-medium').height);
// ##### rescale coordinates vectors
//coordX=(mapWidth/1000)*coordX; // 1.65 * coordX (1.65 times bigger to get to current scale)
//coordY=(mapHeight/677)*coordY; // 1.65 * coordY
map=document.getElementById('mapzoom2');
parentWidth=map.parentNode.offsetWidth;
parentHeight=map.parentNode.offsetHeight;
// calculate target box width and position
//boxWidth=(mapWidth/1000)*targetBoxWidth;
//boxHeight=(mapHeight/677)*targetBoxHeight;
boxWidth = (mapWidth / 1650) * targetBoxWidth;
boxHeight = (mapHeight / 1100) * targetBoxHeight;
boxLeft=coordX;
boxTop = coordY;
// add in header height * 2 to offset the top in viewports with different aspect ratios to map
mapLeft = parseInt(Math.round(coordX - ((parentWidth)/2)+(boxWidth/2)));
mapTop = parseInt((Math.round(coordY - (parentHeight)/(parentWidth/parentHeight)+(parentHeight/4)))-(document.getElementById('header').offsetHeight*2));
// calculate the XY coords we want to move too
xMoveTo = parseInt("-"+mapLeft);
yMoveTo = parseInt("-"+mapTop);
// deal with negatives
if(isNaN(xMoveTo)){
xMoveTo = 20;
}
if(isNaN(yMoveTo)){
yMoveTo = 20;
}
// turn off search, as we're already scrolling
allowSearch=false;
// calculate amount of distance we're going
offSetLeft = document.getElementById('tubemap').offsetLeft;
offSetTop = document.getElementById('tubemap').offsetTop;
totalMoveByX=parseInt((dd.elements.mapzoom2.x-offSetLeft)-xMoveTo);
totalMoveByY=parseInt((dd.elements.mapzoom2.y-offSetTop)-yMoveTo);
// round out the journey -- otherwise long journeys go too fast so need more steps in between
if(Math.abs(totalMoveByX) >= 300 || Math.abs(totalMoveByY) >= 300) {
totalSteps=50;
} else if(Math.abs(totalMoveByX) >= 200 || Math.abs(totalMoveByY) >= 200) {
totalSteps=40;
} else {
totalSteps=20;
}
// actual amount to move in each call from the setInterval
totalMoveByXIncrement=Math.round(totalMoveByX/totalSteps);
totalMoveByYIncrement=Math.round(totalMoveByY/totalSteps);
// set negative/position movement
if(totalMoveByXIncrement > 0){
totalMoveByXIncrement=parseInt("-"+totalMoveByXIncrement);
} else if (totalMoveByXIncrement < 0) {
totalMoveByXIncrement=Math.abs(totalMoveByXIncrement);
} else {
totalMoveByXIncrement=1;
}
if(totalMoveByYIncrement > 0){
totalMoveByYIncrement=parseInt("-"+totalMoveByYIncrement);
} else if (totalMoveByYIncrement < 0) {
totalMoveByYIncrement=Math.abs(totalMoveByYIncrement);
} else {
totalMoveByYIncrement=1;
}
// set the scroll interval
scrollToTargetInterval=window.setInterval("scrollToTarget()",10);
if(document.all){
setTargetBox();
}
}
// stop form submission
return false;
}
function scrollToTarget(){

allowSearch=true;
scrollCount=0;
document.getElementById('targetBox').style.display="block";

if(!document.all){

fadeBoxIn = setInterval(fadeInTargetBox,7);
fadeCount=1;
}
allowSearch=true;
// scroll in progress
} else {
dd.elements.mapzoom2.moveBy(totalMoveByXIncrement, totalMoveByYIncrement);
scrollCount++;
}
}
